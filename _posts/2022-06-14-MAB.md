## Setting
In a stochastic multi-armed bandit (MAB) problem a gambler needs to choose at each round of play one of k arms, each characterized by an unknown reward distribution. Reward realizations are only observed when an arm is selected, and the gambler’s objective is to maximize the cumulative expected earnings after T rounds. To regulate the problem, we assume that we get a loss instead of a reward at each pull. At round $t$, the player choose an arm $a_i$ and observe a loss $l_t(a_i)$. For each arm $i \in [k]$, $l_1(a_i),l_2(a_i)...l_T(a_i)$ are i.i.d random variables with expectation $\mu_i$. We define regret $R_T$ as:
\begin{equation} 
     R_T=\mathbb{E}[\sum_{t=1}^{T}l_t(a_t)-\mu^*]=\mathbb{E}[\sum_{t=1}^{T}l_t(a_t)]-T*\mu^*
\end{equation}
where $\mu^* =\arg \min_{i\in[k]}\mu_i$, $a_t$ is the arm we choose at round $t$. Our goal is to minimize $R_T$.\\
There are two kinds of strategies when we choose an arm. One is exploitation, which means we choose the arm with the lowest empirical loss so far. However, our information about the distribution of an arm is not sufficient without enough pulls. So the other strategy is exploration, in which we try to pull arms that we haven't pulled enough times and get more information about their distribution.

\subsection{Upper Confidence Bound(UCB) Algorithm}
The basic idea of UCB algorithm is that after $t$ rounds, we have high confidence that the reward(loss) of each arm falls in certain interval. The UCB algorithm is an optimistic strategy, we choose the arm with highest upper bound(lowest lower bound in the setting of loss) of the estimated interval so far. If the chosen arm do present the best reward, then this is an exploitation. Otherwise, it's a kind of exploration, which helps us correct our estimation and make better choice in the future.
The following algorithm is based on the setting of loss.
\begin{algorithm}
    \caption{UCB algorithm}
        $n_t(a)$ represents the times arm $a$ is pulled at time t,  $\hat{\mu_t}(a)$ represents the empirical loss of arm a at time t\\
        \textbf{Initialize:} for all $a\in [k]$, set $n_{0}(a)=0$

        \textbf{For $t=1,2...T$ do}
        \begin{algorithmic}[1]
            \STATE compute $a_{t}=\mathop{argmin}\limits_{a\in[k]} \hat{\mu_{t-1}}(a)-\sqrt{\frac{lnT}{n_{t-1}(a)}}$
            \STATE pull arm $a_{t}$, observe the loss of $a_{t}$
            \STATE update $n_t(a_t),\hat{\mu_t}(a_t)$
        \end{algorithmic}
        \textbf{END}
\end{algorithm}
\subsection{analysis of UCB algorithm}
we consider $\mu_t(a_i)$ is the loss by pulling arm $a_i$ at time t.$a^{*}$is defined as follows:
$$a^{*} = argmin_{i\in [k]}\sum_{t=1}^{T}\mu_t(a_i)$$
which means $a^{*}$ has the lowest average loss.\par
we recall the regret of UCB is:
$$R_T = E[\sum_{t=1}^{T}(\mu_t(a_t))] - \sum_{t=1}^{T}\mu(a^{*})$$
\begin{theorem}
$$R_T\leq \sum_{a:\Delta_a >0}({\frac{16lnT}{\Delta_{a}}} + 2\Delta_{a})$$
where $\Delta_a$ is defined as $\mu_a - \mu_a^{*}$ and refers to the gap of average loss between a and the best one.
\end{theorem}
\begin{proof}
Firstly, note that $$R_T = E[\sum_{t=1}^{T}(\mu_t(a_t)) - \mu(a^{*})] = \sum_{a:\Delta_a >0}\Delta_a \cdot E[n_T(a)] $$
so we only need to show for all suboptimal a($\Delta_a > 0$)
$$E[n_T(a)] \leq O(\frac{lnT}{\Delta_a^2})+2 $$
we first show for $\forall a$ and $\forall n>0$
$$E[n_T(a)] \leq n + \sum_{t=n+1}^{T}P(a_t = a \wedge n_{t-1}(a) > n) $$
in fact
\begin{align}
    E[n_T(a)] = \sum_{t=1}^{T}P(a_t = a) = \sum_{t=1}^{T}P(a_t = a\wedge n_{t-1}(a) \ge n) + \sum_{t=1}^{T}P(a_t = a\wedge n_{t-1}(a) < n)
\end{align}
now we separately calculate this two expressions

\begin{align}
    \sum_{t=1}^{T}P(a_t = a\wedge n_{t-1}(a) < n)
    &=\sum_{t=1}^{T}E{{I}}[a_t = a\wedge n_{t-1}(a) < n]\notag\\
    &=E\sum_{t=1}^{T}{{I}}[a_t = a\wedge n_{t-1}(a) < n]\notag\\
    &\leq n\notag
\end{align}
we next bound $\sum_{t=1}^{T}P(a_t = a\wedge n_{t-1}(a) \ge n)$.\par
\begin{proposition}
let's see if $a_t = a$ then one of the following two events must occur
\begin{enumerate}[\bf Event 1:]
    \item $\hat{\mu}_{t-1}(a^*) - \sqrt{\frac{lnT}{n_{t-1}(a^*)}} \geq \mu(a^*)$
    \item $\hat{\mu}_{t-1}(a) - \sqrt{\frac{lnT}{n_{t-1}(a)}} \leq \mu(a^*) = \mu(a) - \Delta_a$
\end{enumerate}
\end{proposition}
that is an obvious fact because if neither of these two occurs ,we have
$\hat{\mu}_{t-1}(a) - \sqrt{\frac{lnT}{n_{t-1}(a)}} > \mu(a^*) > \hat{\mu}_{t-1}(a^*) - \sqrt{\frac{lnT}{n_{t-1}(a^*)}}$then by UCB algorithm we won't choose a at time t. \par
from the proposition above we have
\begin{align}
   P(a_t = a\wedge n_{t-1} \ge n)
    &\leq P(\hat{\mu}_{t-1}(a^*) - \sqrt{\frac{lnT}{n_{t-1}(a^*)}}\geq \mu(a^*)\wedge n_{t-1}(a) > n)\notag\\
    &+P(\hat{\mu}_{t-1}(a) - \sqrt{\frac{lnT}{n_{t-1}(a)}} \leq \mu(a) - \Delta_a\wedge n_{t-1}(a) > n)\notag\\
    &\leq P(\hat{\mu}_{t-1}(a^*) - \sqrt{\frac{lnT}{n_{t-1}(a^*)}}\geq \mu(a^*))\notag\\
    &+P(\hat{\mu}_{t-1}(a) - \sqrt{\frac{lnT}{n_{t-1}(a)}} \leq \mu(a) - \Delta_a \wedge n_{t-1}(a)>n)\notag
\end{align}
we use Chernoff bound in the following process\par
For event 1:
\begin{align}
    P(\hat{\mu}_{t-1}(a^*) - \sqrt{\frac{lnT}{n_{t-1}(a^*)}}\geq \mu(a^*))
    &= \sum_{\tau = 1}^{T}P(\hat{\mu}_{t-1}(a^*) - \sqrt{\frac{lnT}{n_{t-1}(a^*)}}\geq \mu(a^*), n_{t-1}(a^*) = \tau)\notag\\
    &\leq \sum_{\tau = 1}^{T}P(\hat{\mu}_{t-1}(a^*) - \sqrt{\frac{lnT}{n_{t-1}(a^*)}}\geq \mu(a^*)|n_{t-1}(a^*) = \tau)\notag\\
    &\leq \sum_{\tau = 1}^{T}P(\hat{\mu}_{t-1}(a^*) - \mu(a^*)\geq \sqrt{\frac{lnT}{n_{t-1}(a^*)}}|n_{t-1}(a^*) = \tau)\notag\\
    &\leq \sum_{\tau=1}^{T}exp(-2\tau(\frac{lnT}{\tau}))\notag\\
    &=\frac{1}{T}\notag
\end{align}
For event 2: since the inequality holds for all $n>0$,we choose $n = O(\frac{lnT}{\Delta_{a}^{2}})$ such that $\Delta_a = 2\sqrt{\frac{lnT}{n}}$ ,so similarly with event 1 we have
\begin{align}
    P(\hat{\mu}_{t-1}(a) - \sqrt{\frac{lnT}{n_{t-1}(a)}} \leq \mu(a) - \Delta_a \wedge n_{t-1}(a)>n)
    &\leq P(\hat{\mu}_{t-1}(a) - \sqrt{\frac{lnT}{n_{t-1}(a)}} \leq \mu(a) - \Delta_a)\notag\\
    &=P(\mu(a)-\hat{\mu}_{t-1}(a)\ge\Delta_a-\sqrt{\frac{lnT}{n_{t-1}(a)}})\notag\\
    &=P(\mu(a)-\hat{\mu}_{t-1}(a)\ge 2\sqrt{\frac{lnT}{n}}-\sqrt{\frac{lnT}{n_{t-1}(a)}})\notag\\
    &\le P(\mu(a)-\hat{\mu}_{t-1}(a)\ge \sqrt{\frac{lnT}{n_{t-1}(a)}})\notag\\
    &\leq \sum_{\tau=1}^{T}exp(-2\tau(\frac{lnT}{\tau}))\notag\\
    &=\frac{1}{T}\notag
\end{align}
In conclusion, we come back to (10.1)$E[n_T(a)] \leq n+ T\times (\frac{1}{T} + \frac{1}{T}) = O(\frac{lnT}{\Delta_{a}^{2}}) + 2$\par
and so we have $R_T = \sum\limits_{a:\Delta_a >0}\Delta_a \cdot E[n_T(a)] \leq \sum\limits_{a:\Delta_a >0}({\frac{16lnT}{\Delta_{a}}} + 2\Delta_{a})$
\end{proof}
\begin{theorem}
Assume $\Delta_a$ is independent on T.The worst case of $R_T$ of UCB is $O(\sqrt{T\cdot klnT})$
\end{theorem}
\begin{proof}
Divide the suboptimal arms into two groups:
\begin{itemize}
    \item $\Delta_a \leq \sqrt{\frac{klnT}{T}}$
    \item $\Delta_a > \sqrt{\frac{klnT}{T}}$
\end{itemize}
In the first group each a can at most contribute $\Delta_a$ regret,so the whole group at most contribute $T \cdot \Delta_a$.We use Theorem 1 to estimate the second group.

so $R_T$ call be writen as
\begin{align}
    R_T &=\sum\limits_{a:\Delta_a >0}\Delta_a \cdot E[n_T(a)] \notag\\
    &=\sum\limits_{a:0 < \Delta_a \leq \sqrt{\frac{klnT}{T}}}\Delta_a \cdot E[n_T(a)]+\sum\limits_{a:\Delta_a > \sqrt{\frac{klnT}{T}}}\Delta_a \cdot E[n_T(a)]\notag\\
    &\leq T\cdot \sqrt{\frac{klnT}{T}} + \sum_{a:\Delta_a > \sqrt{\frac{klnT}{T}}}({\frac{16lnT}{\Delta_{a}}} + 2\Delta_{a})\notag\\
    &\leq \sqrt{T\cdot klnT} +O(k\cdot \frac{lnT}{\sqrt{\frac{klnT}{T}}})\notag\\
    &=O(\sqrt{T\cdot klnT})\notag
\end{align}
the first inequality holds because of the reason above, the second inequality holds because we assume $\Delta_a$ is independent on T, which is similar to assume $\Delta_a$ is bounded, so it can be ignored when we have lnT.
\end{proof}

\section{Thompson Sampling}
Thompson sampling is an algorithm with randomness. We assume arm $a$ to have a beta prior $Beta(S_a+1,F_a+1)$, where $S_a$ represents times of success, and $F_a$ times of failure.
\begin{definition}[Beta distribution]
For $Beta(\alpha, \beta)(\alpha>0, \beta>0)$, the probability density function is given by:
\begin{equation}
    f(x, \alpha, \beta)=\frac{\Gamma(\alpha+\beta)}{\Gamma(\alpha) \Gamma(\beta)} x^{\alpha-1}(1-x)^{\beta-1}
\end{equation}
\end{definition}

At time $t$, the algorithm samples all of the arms, getting $\Theta_a$ from each $Beta(S_a+1,F_a+1)$. Then we choose the arm that has largest $\Theta_a$, play it, and get reward $r_t(a_t)$. If $r_t(a_t)\notin \{0,1\}$, the algorithm tosses a $r_t(a_t)$-coin to get a \{0,1\}-result, where 1 represents success and 0 failure. Finally, we update $S_a$ and $F_a$, then go to time $t+1$.

\begin{algorithm}
    \caption{Thompson Sampling}
    \begin{algorithmic}[1]
        \STATE \textbf{Initialization}: $a=1,2 \cdots K, S_{a}=0, F_{a}=0$
        \STATE \textbf{for} $t=1,2, \ldots, T$ \textbf{do}
        \STATE $\quad$ For each arm $a$, select $\Theta_{a}(t) \sim \operatorname{Beta}\left(S_{a}+1, F_{a}+1\right)$
        \STATE $\quad$ Play $\operatorname{arm} a_{t}=\arg \max _{a} \Theta_{a}(t)$ and get reward $r_{t}\left(a_{t}\right) \in[0,1]$
        \STATE $\quad$ Toss a $r_{t}\left(a_{t}\right)$-coin, get $\widetilde{r_{t}}\left(a_{t}\right) \in\{0,1\}$
        \STATE $\quad$ \textbf{If} $\widetilde{r_{t}}\left(a_{t}\right)=1, S_{a} \leftarrow S_{a}+1 ;$ \textbf{Else}, $F_{a} \leftarrow F_{a}+1$
    \end{algorithmic}
\end{algorithm}
\begin{theorem}
For $\forall \epsilon>0$, Thompson sampling has regret:
    \begin{equation}
        R_{T} \leq(1+\epsilon) \sum_{a \neq a^{*}} \frac{\log T \Delta_{a}}{d\left(\mu_{a}, \mu_{a^{*}}\right)}+O\left(\frac{k}{\epsilon^{2}}\right)
    \end{equation}
where $d(u, v)=u \ln \frac{u}{v}+(1-u) \ln \frac{1-u}{1-v}$
\end{theorem}
\begin{proof}
We notice that we have the equation
\begin{equation}
    R_{T}=E_{\mathcal{A}} \sum_{t=1}^{T} l_{t}\left(a_{t}\right)-\sum_{t=1}^{T} \mu_{1}=\sum_{a=1}^{k} \Delta_{a} E_{\mathcal{A}}\left[n_{T}(a)\right]
\end{equation}
Now we need to bound each arm's $E_{\mathcal{A}}\left[n_{T}(a)\right]$ to prove our theorem. It's quite complex so you can refer to [1] to see the detailed calculation:
$$
E_{\mathcal{A}}\left[n_{T}(a)\right] \leq\left(1+\epsilon_{0}\right)^{2} \frac{\log T}{d\left(\mu_{a}, \mu_{a^{*}}\right)}+O\left(\frac{1}{\epsilon_{0}^{2}}\right)
$$
And then we take $\epsilon=3 \epsilon_{0}$,
$$
R_{T} \leq \sum_{a \neq a^{*}}\left(1+\epsilon_{0}\right)^{2} \frac{\log T \Delta_{a}}{d\left(\mu_{a}, \mu_{a^{*}}\right)}+O\left(\frac{1}{\epsilon_{0}^{2}}\right) \leq(1+\epsilon) \sum_{a \neq a^{*}} \frac{\log T \Delta_{a}}{d\left(\mu_{a}, \mu_{a^{*}}\right)}+O\left(\frac{k}{\epsilon^{2}}\right)
$$
\end{proof}
